<script>

let devices=[];
let currentTab="all";
let programaPendiente=null;
let rutaPendiente=null;

async function loadDevices(){

 const response=
 await fetch("http://192.168.0.97:8000/devices");

 devices=
 await response.json();

 renderDevices();

}

function cambiarTab(tab,event){

 currentTab=tab;

 document
 .querySelectorAll(".tab")
 .forEach(t=>t.classList.remove("active"));

 event.target.classList.add("active");

 renderDevices();

}

function renderDevices(){

 const grid=
 document.getElementById("device-grid");

 grid.innerHTML="";

 const online=
 devices.filter(d=>d.os);

 let lista=[];

 if(currentTab==="all"){
   lista=devices;
 }

 if(currentTab==="lab1"){
   lista=online.slice(0,2);
 }

 if(currentTab==="lab2"){
   lista=online.slice(2,3);
 }

if(currentTab==="lab3"){
   lista=online.slice(3,4);
 }

 lista.forEach(device=>{

 const isOnline=device.os!==null;

 const card=document.createElement("div");

 card.className="device-card";

 card.innerHTML=`

 <div><strong>${device.hostname||"Sin nombre"}</strong></div>

 <div class="status ${isOnline?"online":"offline"}">
 ${isOnline?"ONLINE":"OFFLINE"}
 </div>

 <div class="device-info">
 ID: ${device.id}
 </div>

 <div class="device-info">
 IP: ${device.ip}
 </div>

 <div class="device-info">
 Sistema:
 ${device.os||"No disponible"}
 </div>

 <div class="device-info">
 Tipo:
 ${device.tipo_dispositivo||"N/A"}
 </div>

 <div class="device-info">
 Última conexión:
 <br>
 ${device.last_seen||"N/A"}
 </div>

 <button
 class="btn btn-history"
 onclick="event.stopPropagation();verHistorial(${device.id})">
 Ver historial
 </button>

 `;

 card.onclick=()=>loadSoftware(
 device.id,
 device.hostname
 );

 grid.appendChild(card);

 });

}

async function loadSoftware(id,nombre){

 document.getElementById(
 "software-panel"
 ).style.display="block";

 document.getElementById(
 "panel-title"
 ).textContent=
 "Administrar equipo: "+nombre;

 const response=

await fetch(
 `http://192.168.0.97:8000/software/${id}`
 );

 const software=
 await response.json();

 const tbody=
 document.getElementById(
 "installed-software"
 );

 tbody.innerHTML="";

 software.forEach(p=>{

 tbody.innerHTML+=`

 <tr>
 <td>${p.name||"N/A"}</td>
 <td>${p.version||"N/A"}</td>

 <td>

 <button
 class="btn btn-remove"
 onclick="desinstalarPrograma(
 ${p.device_id},
 '${(p.name||"").replace(/'/g,"\\'")}'
 )">

 Desinstalar

 </button>

 </td>

 </tr>

 `;

 });

}

function abrirSelector(programa,ruta){

 programaPendiente=programa;
 rutaPendiente=ruta;

 const online=
 devices.filter(d=>d.os);

 const lista=
 document.getElementById("listaEquipos");

 lista.innerHTML="";

 online.forEach(d=>{

 lista.innerHTML+=`

 <div>
 <label>
 <input type="checkbox" value="${d.id}">
 ${d.hostname} (${d.ip})
 </label>
 </div>

 `;

 });

 document.getElementById(

 "selectorEquipos"
 ).style.display="block";

}

function cerrarSelector(){

 document.getElementById(
 "selectorEquipos"
 ).style.display="none";

}

async function instalarSeleccionados(){

 const checks=
 document.querySelectorAll(
 "#listaEquipos input:checked"
 );

 for(const c of checks){

 await fetch(
 "http://192.168.0.97:8000/acciones",
 {
 method:"POST",
 headers:{
 "Content-Type":"application/json"
 },
 body:JSON.stringify({
 device_id:Number(c.value),
 accion:"instalar",
 programa:programaPendiente,
 ruta_instalador:rutaPendiente
 })
 });

 }

 cerrarSelector();

 alert("Solicitud enviada");

}

async function desinstalarPrograma(
 deviceId,
 programa
){

 const confirmar=
 confirm(
 `¿Seguro que deseas desinstalar "${programa}"?`
 );

 if(!confirmar){
   return;
 }

 await fetch(
 "http://192.168.0.97:8000/acciones",
 {
 method:"POST",
 headers:{
 "Content-Type":"application/json"
 },
 body:JSON.stringify({
 device_id:Number(deviceId),
 accion:"desinstalar",
 programa:programa,
 ruta_instalador:""
 })
 });

 alert("Solicitud de desinstalación enviada");

}

async function verHistorial(deviceId){

 const panel=
 document.getElementById("historialPanel");

 const contenido=
 document.getElementById("historialContenido");

 contenido.innerHTML="Cargando historial...";

 panel.style.display="block";

 try{

   const response=
   await fetch(
   `http://192.168.0.97:8000/historial/${deviceId}`
   );

   const historial=
   await response.json();

   if(!historial.length){

 	contenido.innerHTML=
 	"Sin historial registrado.";

 	return;
   }

   contenido.innerHTML=
   historial.map(h=>`

   <div style="margin-bottom:10px;border-bottom:1px solid #ddd;padding-bottom:8px;">

   <strong>${h.accion}</strong>
   - ${h.programa}

   <br>

   ${h.fecha}

   </div>

   `).join("");

 }catch{

   contenido.innerHTML=
   "No fue posible cargar el historial.";

 }

}

function cerrarHistorial(){

 document.getElementById(
 "historialPanel"
 ).style.display="none";

}

loadDevices();

setInterval(loadDevices,30000);
